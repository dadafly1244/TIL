let colorEl = Math.random(); //0.44904594945086096
colorEl = Math.random().toString(16); // 16진수로 만들기 0.4dd0e9397816f
colorEl = Math.random().toString(16).substring(2, 8); // 2번 인덱스 부터 8전까지 
console.log(colorEl);
